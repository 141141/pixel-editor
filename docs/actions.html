module.exports = Actions = (I={}, self=Core(I)) ->
  self.extend
    addAction: (action) ->
      self.actions.push action

    actions: Observable []

  Object.keys(Actions.defaults).forEach (hotkey) ->
    {method, icon} = Actions.defaults[hotkey]

    self.addAction
      perform: ->
        if typeof method is "function"
          method
            editor: self
        else
          self[method]()
      iconUrl: icon

    self.addHotkey hotkey, method

  return self

state = null
